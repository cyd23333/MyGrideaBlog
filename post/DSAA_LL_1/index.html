#include <stdlib.h>

typedef struct LNode{
    float c;
    int e;
    int length;
    struct LNode* next;
}LNode;

LNode* LLInit(){
    //Func 0
    //Create an empty Head Node
    LNode* LH = (LNode*)malloc(sizeof(LNode));
    LH->next  = (LNode*)malloc(sizeof(LNode));
    LH->length = 0;
    return LH;
}

void LLReadCreate(FILE* fp, LNode* LHA);// FUNC A
void LLSort(LNode* LHB);// FUNC B

void LLReadCreate(FILE* fp, LNode* LHA){
    //Func A
    //Reading from text and then create LL.....
    float coef = 0.0; int expf = 0;
    LHA->length = 0;
    LNode* p, *ptr;
    p = LHA;
    while(1){
        fscanf(fp, "%f %d", &coef, &expf);
        if(coef==0 && expf==0){
            if(p == LHA){
                ptr = (LNode*)malloc(sizeof(LNode));
                ptr->c = ptr->e = 0;
                p->next = ptr;  ptr->next = NULL;
                LHA->length = 1;
                break;
            }
            else{
                p->next = NULL;
                break;
            }
        }// (0,0)
        if(coef != 0){
            p->next = (LNode*)malloc(sizeof(LNode));
            ptr = p;
            p = p->next;
            p->c = coef;    p->e = expf;    LHA->length ++;
        }
    }
    printf("Done. Read and Created LL from the text file.\n");
}

void LLSort(LNode* LHB){
    //Func B
    //Descending Order
    //// Considering using <stdlib.h> qsort()
    if(LHB->length == 0){
        printf("The List is EMPTY! Please check!!\n");
        return;
    }
    if(LHB->length == 1)   return;

    LNode *p, *q, *ptr;
    p = LHB->next;   q = p->next;

    float coef = 0; int expf = 0;
    while(1){
        while(q != NULL){
            if (p->e < q->e){
                coef = p->c;  expf = p->e;
                p->c = q->c;  p->e = q->e;
                q->c = coef;  q->e = expf;
            }
            else if(p->e == q->e){
                p->c = p->c + q->c;
                ptr->next = q->next;
                LHB->length --;
            }
            ptr = q;
            q = q->next;
        }
        p = p->next;
        q = p->next;// Once after a SUBCYCLE, reset the position of the ptr* q
    if (p->next == NULL)    break;
    }
}

LNode* LLAdd(LNode* LA, LNode* LB, int op){
    LNode* p, *q, *ptr;
    LNode* LC = LLInit();   LC->length = 0;
    LNode* pc = LLInit();
    p = LA->next;   q = LB->next;
    if ( (LA->length==0 && LB->length==0)  || (LA->next==NULL && LB->next==NULL) ){
        printf("These two LL are ALL EMPTY! Please check!");
        return LC;
    }// Robustness
    pc = LC->next;
    float coef = 0.00;

    while(q!=NULL && p!=NULL){
        if(p->c==0){
            p = p->next;
            continue;
        }
        if(q->c==0){
            q = q->next;
            continue;
        }
        if(p->e < q->e){
            pc->c = q->c;   pc->e = q->e;   
            LC->length ++;
            q = q->next;
            pc->next = (LNode*)malloc(sizeof(LNode));
            ptr = pc;
            pc = pc->next;
        }
        else if(p->e > q->e){
            pc->c = p->c;   pc->e = p->e;
            LC->length ++;
            p = p->next;
            pc->next = (LNode*)malloc(sizeof(LNode));
            ptr = pc;
            pc = pc->next;
        }
        else{
            if (op == 1)    coef = p->c + q->c;
            if (op == 2)    coef = p->c - q->c;
            if(abs(coef) < 1e-6){
                p = p->next;    q = q->next;
            }
            else{
                pc->c = coef;   pc->e = p->e;
                LC->length ++;
                p = p->next;        q = q->next;
                pc->next = (LNode*)malloc(sizeof(LNode));
                ptr = pc;
                pc = pc->next;
            }
        }
    } // End of while(q!=NULL && p!=NULL)
    if (p == NULL){
        while(q != NULL){
            pc->c = q->c;   pc->e = q->e;  LC->length ++;
            q = q->next;
            ptr = pc;
            pc->next = (LNode*)malloc(sizeof(LNode));
            pc = pc->next;
        }
    }
    if (q == NULL){
        while(p != NULL){
            pc->c = p->c;   pc->e = p->e;  LC->length ++;
            p = p->next;
            ptr = pc;
            pc->next = (LNode*)malloc(sizeof(LNode));
            pc = pc->next; 
        }
    }
    ptr->next = NULL;
    free(pc);
    return LC;
}

LNode* LLDiff(LNode* LHE){
    LNode* p, *ptr, *LE, *pe;
    LE = LLInit();
        pe = LE->next;
    p = LHE;

    if(p->next == NULL){
        printf("\nCurrent LL is EMPTY!\n");
        return NULL;
    }
    while(p->next != NULL){    
        p = p->next;
        if(p->e == 0)   continue;
        pe->c = p->c*p->e;
        pe->e = p->e - 1;
        LE->length ++;
        pe->next = (LNode*)malloc(sizeof(LNode));
        ptr = pe;
        pe = pe->next;
    }
    ptr->next = NULL;
    return LE;
}

void LLPrint(LNode* LHF){
    //Func F
    //Print as formatted LL
    LNode* p;
    if(LHF->length != 0 || LHF != NULL){
        p = LHF->next;
        printf("This LL is %d long ",LHF->length);
        while (p != NULL){
            printf(",    %.2f,%d ", p->c, p->e);
            p = p->next;         
        }
        printf("\n");
    }
    else    printf("The List is EMPTY!\n");
}

void LLOperate(FILE* fp, LNode* LA, LNode* LB, int Operation){
    // Do some DEFINE: plus 1 // minus 2
    LLReadCreate(fp,LA);    LLReadCreate(fp,LB);
    if(LA->length != 0){
        printf("\nGoing to Print the first LL:\n");   
        LLSort(LA); LLPrint(LA);
    }else    printf("\nThe first LL is EMPTY!! Please check!\n");
    if(LB->length != 0){
        printf("\nGoing to Print the second LL:\n");
        LLSort(LB);  LLPrint(LB);
    }else    printf("\nThe second LL is EMPTY!! Please check!\n");

    LNode* LC = LLAdd(LA,LB,Operation);   
        printf("\nNow We Print the Calculate Result LL:\n");
        LLPrint(LC);
        if(Operation == 1)  printf("P.S.  While we Swap these two poly, they will give the SAME Result\n");
    
    LNode* LD = LLDiff(LA);
    LNode* LE = LLDiff(LB); 
    if(LD->length!=0 && LE->length!=0){
        printf("\nNow We Print the Diff Result of the first LL:\n");    LLPrint(LD);
        printf("\nNow We Print the Diff Result of the second LL:\n");   LLPrint(LE);
    }
    else if(LD->length!=0){
        printf("\nThe second LL is EMPTY.\nNow We Print the Diff Result of the first LL:\n");
        LLPrint(LD);
    }
    else if(LE->length!=0){
        printf("\nThe first LL is EMPTY\nNow We Print the Diff Result of the second LL:\n");
        LLPrint(LE);
    }
    else    printf("\nThe two LL are ALL EMPTY\n");
}
